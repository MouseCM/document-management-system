\begin{frame}{Project Overview}
\begin{itemize}
    \item \textbf{Goal:} Standardize project document storage and release process.
    \item \textbf{Key Features:}
    \begin{itemize}
        \item Version control and storage (MinIO)
        \item ABAC/RBAC Access Control
        \item Audit Logging
        \item Diff highlighting (PDF \& Word)
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Technology Stack}
\begin{itemize}
    \item \textbf{Backend:} Node.js (Express \& TypeScript)
    \item \textbf{Frontend:} React (Vite)
    \item \textbf{Database:} PostgreSQL (Prisma ORM)
    \item \textbf{Storage:} MinIO
    \item \textbf{Deployment:} Docker \& Docker Compose
\end{itemize}
\end{frame}

\begin{frame}{Security \& Compliance}
\begin{itemize}
    \item \textbf{Encryption:} Server-Side Encryption (SSE) applied to all MinIO objects.
    \item \textbf{Audit Log:} Append-only PostgreSQL table tracking CREATE, UPLOAD, VIEW, DOWNLOAD, and EDIT actions.
    \item \textbf{Access Control:} Inherited roles from Department to Project (Viewer, Editor, Admin).
\end{itemize}
\end{frame}

\begin{frame}{Document Processing}
\begin{itemize}
    \item \textbf{Max File Size:} 5MB limit enforced.
    \item \textbf{Retention Policy:} Configurable cleanup background cron job.
    \item \textbf{Diff Engine:}
    \begin{itemize}
        \item \textit{PDF:} Text extraction and word-level diff.
        \item \textit{Word:} \texttt{document.xml} extraction and XML tree diffing.
    \end{itemize}
\end{itemize}
\end{frame}
